620 if (fabs(Jij)>1e-10) {
621 // 1) J_kl (-icos + sin) * _{Ek+l i, i}
622 if (i > 0 && j < nj-1) {
623 int itx = it - stridei + stridej;
624 double xre = xptr[2 * itx];
625 double xim = xptr[2 * itx + 1];
626 double sq = sqrt(i * (j + 1));

638 *yre += Jij * sq * ( cosij * xim - sinij * xre);

649 *yim += Jij * sq * ( cosij * xre + sinij * xim);

801inline void control(const int it, const int n, const int i, const int np, const int ip, const int stridei, const int strideip, const double* xptr, const double pt, const double qt, double* yre, double* yim){
802 /* \rho(ik+1..,ik'..) term */
803 if (i < n-1) {
804 int itx = it + stridei;
805 double xre = xptr[2 * itx];
806 double xim = xptr[2 * itx + 1];
807 double sq = sqrt(i + 1);
808 *yre += sq * ( pt * xim + qt * xre);
809 *yim += sq * ( - pt * xre + qt * xim);
810 }
811 /* \rho(ik..,ik'+1..) */
812 if (ip < np-1) {
813 int itx = it + strideip;
814 double xre = xptr[2 * itx];
815 double xim = xptr[2 * itx + 1];
816 double sq = sqrt(ip + 1);
817 *yre += sq * ( -pt * xim + qt * xre);
818 *yim += sq * ( pt * xre + qt * xim);
819 }
820 /* \rho(ik-1..,ik'..) */
821 if (i > 0) {
822 int itx = it - stridei;
823 double xre = xptr[2 * itx];
824 double xim = xptr[2 * itx + 1];
825 double sq = sqrt(i);
826 *yre += sq * ( pt * xim - qt * xre);
827 *yim += sq * (- pt * xre - qt * xim);
828 }
829 /* \rho(ik..,ik'-1..) */
830 if (ip > 0) {
831 int itx = it - strideip;
832 double xre = xptr[2 * itx];
833 double xim = xptr[2 * itx + 1];
834 double sq = sqrt(ip);
835 *yre += sq * (- pt * xim - qt * xre);
836 *yim += sq * ( pt * xre - qt * xim);
