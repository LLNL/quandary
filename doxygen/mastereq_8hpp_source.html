572inline void Jkl_coupling(const int it, const int ni, const int nj, const int nip, const int njp, const int i, const int ip, const int j, const int jp, const int stridei, const int strideip, const int stridej, const int stridejp, const double* xptr, const double Jij, const double cosij, const double sinij, double* yre, double* yim) {
573 if (fabs(Jij)>1e-10) {
574 // 1) J_kl (-icos + sin) * ρ_{E−k+l i, i′}
575 if (i > 0 && j < nj-1) {
576 int itx = it - stridei + stridej;
577 double xre = xptr[2 * itx];
578 double xim = xptr[2 * itx + 1];
579 double sq = sqrt(i * (j + 1));
580 // sin u + cos v + i ( -cos u + sin v)
581 *yre += Jij * sq * ( cosij * xim + sinij * xre);
582 *yim += Jij * sq * ( - cosij * xre + sinij * xim);
583 }
584 // 2) J_kl (−icos − sin)sqrt(il*(ik +1)) ρ_{E+k−li,i′}
585 if (i < ni-1 && j > 0) {
586 int itx = it + stridei - stridej; // E+k-l i, i'
587 double xre = xptr[2 * itx];
588 double xim = xptr[2 * itx + 1];
589 double sq = sqrt(j * (i + 1)); // sqrt( il*(ik+1))
590 // -sin u + cos v + i (-cos u - sin v)
591 *yre += Jij * sq * ( cosij * xim - sinij * xre);
592 *yim += Jij * sq * ( - cosij * xre - sinij * xim);
593 }
594 // 3) J_kl ( icos + sin)sqrt(ik'*(il' +1)) ρ_{i,E-k+li'}
595 if (ip > 0 && jp < njp-1) {
596 int itx = it - strideip + stridejp; // i, E-k+l i'
597 double xre = xptr[2 * itx];
598 double xim = xptr[2 * itx + 1];
599 double sq = sqrt(ip * (jp + 1)); // sqrt( ik'*(il'+1))
600 // sin u - cos v + i ( cos u + sin v)
601 *yre += Jij * sq * ( - cosij * xim + sinij * xre);
602 *yim += Jij * sq * ( cosij * xre + sinij * xim);
603 }
604 // 4) J_kl ( icos - sin)sqrt(il'*(ik' +1)) ρ_{i,E+k-li'}
605 if (ip < nip-1 && jp > 0) {
606 int itx = it + strideip - stridejp; // i, E+k-l i'
607 double xre = xptr[2 * itx];
608 double xim = xptr[2 * itx + 1];
609 double sq = sqrt(jp * (ip + 1)); // sqrt( il'*(ik'+1))
610 // - sin u - cos v + i ( cos u - sin v)
611 *yre += Jij * sq * ( - cosij * xim - sinij * xre);
612 *yim += Jij * sq * ( cosij * xre - sinij * xim);

754inline void control(const int it, const int n, const int i, const int np, const int ip, const int stridei, const int strideip, const double* xptr, const double pt, const double qt, double* yre, double* yim){
755 /* \rho(ik+1..,ik'..) term */
756 if (i < n-1) {
757 int itx = it + stridei;
758 double xre = xptr[2 * itx];
759 double xim = xptr[2 * itx + 1];
760 double sq = sqrt(i + 1);
761 *yre += sq * ( pt * xim + qt * xre);
762 *yim += sq * ( - pt * xre + qt * xim);
763 }
764 /* \rho(ik..,ik'+1..) */
765 if (ip < np-1) {
766 int itx = it + strideip;
767 double xre = xptr[2 * itx];
768 double xim = xptr[2 * itx + 1];
769 double sq = sqrt(ip + 1);
770 *yre += sq * ( -pt * xim + qt * xre);
771 *yim += sq * ( pt * xre + qt * xim);
772 }
773 /* \rho(ik-1..,ik'..) */
774 if (i > 0) {
775 int itx = it - stridei;
776 double xre = xptr[2 * itx];
777 double xim = xptr[2 * itx + 1];
778 double sq = sqrt(i);
779 *yre += sq * ( pt * xim - qt * xre);
780 *yim += sq * (- pt * xre - qt * xim);
781 }
782 /* \rho(ik..,ik'-1..) */
783 if (ip > 0) {
784 int itx = it - strideip;
785 double xre = xptr[2 * itx];
786 double xim = xptr[2 * itx + 1];
787 double sq = sqrt(ip);
788 *yre += sq * (- pt * xim - qt * xre);
789 *yim += sq * ( pt * xre - qt * xim);
790 }
