581 // 1) J_kl (-icos + sin) * _{Ek+l i, i}
582 if (i > 0 && j < nj-1) {
583 int itx = it - stridei + stridej;
584 double xre = xptr[2 * itx];
585 double xim = xptr[2 * itx + 1];
586 double sq = sqrt(i * (j + 1));

588 *yre += Jij * sq * ( cosij * xim + sinij * xre);
589 *yim += Jij * sq * ( - cosij * xre + sinij * xim);

592 if (i < ni-1 && j > 0) {

596 double sq = sqrt(j * (i + 1)); // sqrt( il*(ik+1))
597 // -sin u + cos v + i (-cos u - sin v)
598 *yre += Jij * sq * ( cosij * xim - sinij * xre);
599 *yim += Jij * sq * ( - cosij * xre - sinij * xim);
600 }
601 // 3) J_kl ( icos + sin)sqrt(ik'*(il' +1)) _{i,E-k+li'}

603 int itx = it - strideip + stridejp; // i, E-k+l i'
604 double xre = xptr[2 * itx];
605 double xim = xptr[2 * itx + 1];
606 double sq = sqrt(ip * (jp + 1)); // sqrt( ik'*(il'+1))
607 // sin u - cos v + i ( cos u + sin v)

609 *yim += Jij * sq * ( cosij * xre + sinij * xim);
610 }

762 /* \rho(ik+1..,ik'..) term */
763 if (i < n-1) {
764 int itx = it + stridei;
765 double xre = xptr[2 * itx];
766 double xim = xptr[2 * itx + 1];

769 *yim += sq * ( - pt * xre + qt * xim);

778 *yim += sq * ( pt * xre + qt * xim);

780 /* \rho(ik-1..,ik'..) */
781 if (i > 0) {
782 int itx = it - stridei;
783 double xre = xptr[2 * itx];
784 double xim = xptr[2 * itx + 1];
785 double sq = sqrt(i);
786 *yre += sq * ( pt * xim - qt * xre);
787 *yim += sq * (- pt * xre - qt * xim);
788 }
789 /* \rho(ik..,ik'-1..) */
790 if (ip > 0) {
791 int itx = it - strideip;
792 double xre = xptr[2 * itx];
793 double xim = xptr[2 * itx + 1];
794 double sq = sqrt(ip);
795 *yre += sq * (- pt * xim - qt * xre);
796 *yim += sq * ( pt * xre - qt * xim);
